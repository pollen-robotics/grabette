# Casquette config.txt for Raspberry Pi Zero 2W
#
# Copy to /boot/firmware/config.txt and reboot.
# Also remove "console=serial0,115200" from /boot/firmware/cmdline.txt.
#
# Hardware: camera (CSI) + BMI088 IMU (I2C-1)
# No angle sensors, no servos.
#
# Required env vars (in .env):
#   CASQUETTE_IMU_I2C_BUS=1
#   CASQUETTE_DEVICE_ID=casquette-1

# Enable I2C (hw bus 1 for BMI088 IMU)
dtparam=i2c_arm=on

# Enable audio (loads snd_bcm2835)
dtparam=audio=on

# Automatically load overlays for detected cameras
camera_auto_detect=1

# Automatically load overlays for detected DSI displays
display_auto_detect=1

# Automatically load initramfs files, if found
auto_initramfs=1

# Enable DRM VC4 V3D driver
dtoverlay=vc4-kms-v3d
max_framebuffers=2

# Don't have the firmware create an initial video= setting in cmdline.txt.
disable_fw_kms_setup=1

# Run in 64-bit mode
arm_64bit=1

# Disable compensation for displays with overscan
disable_overscan=1

# Run as fast as firmware / board allows
arm_boost=1
